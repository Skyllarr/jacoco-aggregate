package org.wildfly.security.ssl;

import java.util.EnumSet;

/**
 * The security level setting for SSL/TLS cipher suite selection.
 *
 * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
 */
public enum SecurityLevel {
    NONE,
    EXP40,
    EXP56,
    LOW,
    MEDIUM,
    HIGH,
    FIPS,
    ;
    static final int fullSize = values().length;

    static SecurityLevel forName(final String name) {
        switch (name) {
            case "NONE": return NONE;
            case "EXP40": return EXP40;
            case "EXP56": return EXP56;
            case "LOW": return LOW;
            case "MEDIUM": return MEDIUM;
            case "HIGH": return HIGH;
            case "FIPS": return FIPS;
            default: return null;
        }
    }

    static boolean isFull(final EnumSet<SecurityLevel> levels) {
        return levels != null && levels.size() == fullSize;
    }
}
