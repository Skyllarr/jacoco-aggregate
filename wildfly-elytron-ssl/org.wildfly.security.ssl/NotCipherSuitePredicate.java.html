package org.wildfly.security.ssl;

final class NotCipherSuitePredicate extends CipherSuitePredicate {
    private final CipherSuitePredicate predicate;

    NotCipherSuitePredicate(final CipherSuitePredicate predicate) {
        this.predicate = predicate;
    }

    void toString(final StringBuilder b) {
        b.append("not (");
        predicate.toString(b);
        b.append(")");
    }

    boolean test(final MechanismDatabase.Entry entry) {
        return ! predicate.test(entry);
    }

    boolean isAlwaysTrue() {
        return predicate.isAlwaysFalse();
    }

    boolean isAlwaysFalse() {
        return predicate.isAlwaysTrue();
    }
}
