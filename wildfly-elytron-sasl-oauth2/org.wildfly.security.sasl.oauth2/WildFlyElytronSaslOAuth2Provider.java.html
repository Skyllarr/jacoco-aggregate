package org.wildfly.security.sasl.oauth2;

import java.security.Provider;

import org.kohsuke.MetaInfServices;
import org.wildfly.security.WildFlyElytronBaseProvider;

/**
 * Provider for the OAuth2 SASL authentication mechanism.
 *
 * @author <a href="mailto:fjuma@redhat.com">Farah Juma</a>
 */
@MetaInfServices(Provider.class)
public final class WildFlyElytronSaslOAuth2Provider extends WildFlyElytronBaseProvider {

    private static final long serialVersionUID = 4264694243468176618L;
    private static WildFlyElytronSaslOAuth2Provider INSTANCE = new WildFlyElytronSaslOAuth2Provider();

    /**
     * Construct a new instance.
     */
    public WildFlyElytronSaslOAuth2Provider() {
        super("WildFlyElytronSaslOAuth2Provider", "1.0", "WildFly Elytron SASL OAuth2 Provider");
        putService(new ProviderService(this, SASL_SERVER_FACTORY_TYPE, "OAUTHBEARER",  "org.wildfly.security.sasl.oauth2.OAuth2SaslServerFactory", emptyList, emptyMap, false, true));
        putService(new ProviderService(this, SASL_CLIENT_FACTORY_TYPE, "OAUTHBEARER",  "org.wildfly.security.sasl.oauth2.OAuth2SaslClientFactory", emptyList, emptyMap, false, true));
    }

    /**
     * Get the OAuth2 SASL authentication mechanism provider instance.
     *
     * @return the OAuth2 SASL authentication mechanism provider instance
     */
    public static WildFlyElytronSaslOAuth2Provider getInstance() {
        return INSTANCE;
    }

}
