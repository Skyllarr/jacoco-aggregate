package org.wildfly.security.auth.util;

import org.wildfly.common.Assert;
import org.wildfly.security.auth.server.NameRewriter;

/**
 * A delegating name rewriter with a mutable delegation target.
 *
 * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
 */
public final class MutableNameRewriter implements NameRewriter {
    private volatile NameRewriter target = IDENTITY_REWRITER;

    /**
     * Construct a new instance.
     *
     * @param target the target rewriter (not {@code null})
     */
    public MutableNameRewriter(final NameRewriter target) {
        Assert.checkNotNullParam("target", target);
        this.target = target;
    }

    /**
     * Construct a new instance with the identity rewriter.
     */
    public MutableNameRewriter() {
    }

    public String rewriteName(final String original) {
        return target.rewriteName(original);
    }

    /**
     * Set the target rewriter.
     *
     * @param target the target rewriter (not {@code null})
     */
    public void setTarget(final NameRewriter target) {
        Assert.checkNotNullParam("target", target);
        this.target = target;
    }
}
