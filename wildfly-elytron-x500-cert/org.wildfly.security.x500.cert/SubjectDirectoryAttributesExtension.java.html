package org.wildfly.security.x500.cert;

import java.util.List;

import org.wildfly.common.Assert;
import org.wildfly.security.asn1.ASN1Encoder;
import org.wildfly.security.x500.X500;
import org.wildfly.security.x500.X500DirectoryAttribute;

/**
 * The subject directory attributes extension as defined by <a href="https://tools.ietf.org/html/rfc5280#section-4.2.1.8">RFC 5280 ยง 4.2.1.8</a>.
 *
 * @author <a href="mailto:david.lloyd@redhat.com">David M. Lloyd</a>
 */
public final class SubjectDirectoryAttributesExtension extends X509CertificateExtension {
    private final List<X500DirectoryAttribute> directoryAttributes;

    /**
     * Construct a new instance.
     *
     * @param directoryAttributes the list of directory attributes (must not be {@code null})
     */
    public SubjectDirectoryAttributesExtension(final List<X500DirectoryAttribute> directoryAttributes) {
        super(false);
        Assert.checkNotNullParam("directoryAttributes", directoryAttributes);
        this.directoryAttributes = directoryAttributes;
    }

    public String getId() {
        return X500.OID_CE_SUBJECT_DIRECTORY_ATTRIBUTES;
    }

    public void encodeTo(final ASN1Encoder encoder) {
        encoder.startSequence();
        for (X500DirectoryAttribute directoryAttribute : directoryAttributes) {
            directoryAttribute.encodeTo(encoder);
        }
        encoder.endSequence();
    }
}
